      data-slide

     const swiper = new Swiper(".swiper", {
        loop: true,

        // If we need pagination
        pagination: {
          el: ".swiper-pagination",
        },

        // Navigation arrows
        navigation: {
          nextEl: ".swiper-button-next",
          prevEl: ".swiper-button-prev",
        },

        // And if we need scrollbar
        scrollbar: {
          el: ".swiper-scrollbar",
        },
      });
      // Handle button clicks
      document.querySelectorAll(".go-to-slide").forEach((button) => {
        button.addEventListener("click", () => {
          const index = parseInt(button.dataset.slide, 0);
          swiper.slideTo(index); // or mySwiper.slideToLoop(index) if using loop mode
        });
      });
